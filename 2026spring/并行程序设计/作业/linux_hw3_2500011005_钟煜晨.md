```sh
CUR_DIR=$(pwd)

#统计文件数
CPP_CNT=$(find "$CUR_DIR" -name "*.cpp" | wc -l)
H_CNT=$(find "$CUR_DIR" -name "*.h" | wc -l)

#统计总行数
CPP_TOT=$(find "$CUR_DIR" -name "*.cpp" | xargs cat | wc -l)
H_TOT=$(find "$CUR_DIR" -name "*.h" | xargs cat | wc -l)

#统计总注释数，cat用于合并流
CPP_COMM=$(find "$CUR_DIR" -name "*.cpp" | xargs cat | grep -cE "^[[:space:]]*//|^[[:space:]]*/\*|^[[:space:]]*\*|^[[:space:]]*\*/")

H_COMM=$(find "$CUR_DIR" -name "*.h" | xargs cat | grep -cE "^[[:space:]]*//|^[[:space:]]*/\*|^[[:space:]]*\*|^[[:space:]]*\*/")

#计算注释率
CPP_COMM_RATE=$(echo "scale=2; $CPP_COMM*100 / $CPP_TOT" | bc)
H_COMM_RATE=$(echo "scale=2; $H_COMM*100 / $H_TOT" | bc)

#输出
echo ".cpp文件数量：$CPP_CNT"
echo "总行数：$CPP_TOT"
echo "注释行数：$CPP_COMM"
echo "注释率：$CPP_COMM_RATE %
"

echo ".h文件数量：$H_CNT"
echo "总行数：$H_TOT"
echo "注释行数：$H_COMM"
echo "注释率：$H_COMM_RATE %"

```

## 题目五
![[Pasted image 20260316124757.png]]

>我自己的递归代码跑不出结果
>想必是abacus的头文件有循环引用和性能问题，所以用AI改了一份能跑通的交作业

```sh
#!/bin/bash

# 1. 一次性建立路径索引（存入关联数组，搜索时间复杂度 O(1)）
declare -A FILE_MAP
echo "正在索引文件..."
while read -r path; do
    name=$(basename "$path")
    FILE_MAP["$name"]="$path"
done < <(find . -type f \( -name "*.h" -o -name "*.cpp" \))

# 2. 定义缓存和递归函数
declare -A DEPTH_CACHE
declare -A VISITING  # 防止死循环

get_depth() {
    local file_path=$1
    local file_name=$(basename "$file_path")

    # 如果有缓存直接返回
    [[ -n "${DEPTH_CACHE[$file_path]}" ]] && { echo "${DEPTH_CACHE[$file_path]}"; return; }
    
    # 检测死循环
    [[ -n "${VISITING[$file_path]}" ]] && { echo 0; return; }
    VISITING["$file_path"]=1

    local max_d=0
    # 使用内建 read 循环，避免调用 grep/awk 进程
    while read -r line; do
        # Bash 原生正则匹配 #include "..."
        if [[ $line =~ ^[[:space:]]*#include[[:space:]]*\"([^\"]+)\" ]]; then
            local inc_name="${BASH_REMATCH[1]}"
            local inc_path="${FILE_MAP[$inc_name]}"
            
            if [[ -n "$inc_path" && -f "$inc_path" ]]; then
                local d=$(get_depth "$inc_path")
                (( d + 1 > max_d )) && max_d=$((d + 1))
            fi
        fi
    done < "$file_path"

    unset VISITING["$file_path"]
    DEPTH_CACHE["$file_path"]=$max_d
    echo "$max_d"
}

# 3. 执行统计
echo -e "深度\t路径"
for f in "${FILE_MAP[@]}"; do
    d=$(get_depth "$f")
    echo -e "$d\t$f"
done | sort -nr | head -n 100
```
