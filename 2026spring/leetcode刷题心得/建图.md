## [[链式前向星]]
---
```cpp
struct Edge { 
	int to, next; 
} edge[MAXE]; 

int head[MAXN], cnt; 
void add_edge(int u, int v) { 
	edge[++cnt] = {v, head[u]}; 
	head[u] = cnt; 
}
```

## 有边长
---
```cpp
const int MAXN = 100005; // 每个 vector 存储一个 pair：{邻居节点, 边长} 
vector<pair<int, int>> adj[MAXN]; 

void add_edge(int u, int v, int w) { 
	adj[u].push_back({v, w}); // 将 v 和权重 w 作为一个整体存入 
	}
```

```cpp
struct Edge {
    int to, next, w; // to: 终点, next: 同起点的下一条边, w: 权重
} edge[MAXE];

int head[MAXN], cnt;

void add_edge(int u, int v, int w) {
    edge[++cnt].to = v;
    edge[cnt].w = w;
    edge[cnt].next = head[u];
    head[u] = cnt;
}
```
